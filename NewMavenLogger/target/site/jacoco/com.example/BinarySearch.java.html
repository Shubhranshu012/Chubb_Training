package com.example;


public class BinarySearch {

    boolean search(int[] arr, int target) {
        int left = 0;
        int right = arr.length - 1;

        while (left <= right) {
            int mid = left + (right - left) / 2;

            if (arr[mid] == target) {
                return true; 
            } else if (arr[mid] < target) {
                left = mid + 1; 
            } else {
                right = mid - 1;
            }
        }

        return false;
    }
}
